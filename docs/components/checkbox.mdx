## Usage
<Tabs>
    <Tab title="Default">
        ```tsx
        import {Checkbox} from "@lib/Checkbox";

        <Checkbox/>
        ```
    </Tab>
    <Tab title="Group">
        ```tsx
        import {Checkbox, CheckboxGroup} from "@lib/Checkbox";

        <CheckboxGroup>
            <Checkbox value="1">Text 1</Checkbox>
            <Checkbox value="2">Text 2</Checkbox>
        </CheckboxGroup>
        ```
    </Tab>
    <Tab title="Boxed Group">
        ```tsx
        import {BoxedCheckboxGroup, Checkbox} from "@lib/Checkbox";

        <BoxedCheckboxGroup>
            <Checkbox value="1">Text 1</Checkbox>
            <Checkbox value="2">Text 2</Checkbox>
        </BoxedCheckboxGroup>
        ```
    </Tab>
</Tabs>
