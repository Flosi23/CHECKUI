## Usage

<Tabs>
    <Tab title="Default">
        ```tsx
        import {RadioGroup, RadioGroupItem} from "@/components/ui/Checkbocx";

        <RadioGroup direction="column">
            <RadioGroupItem value="1">Option 1</RadioGroupItem>
            <RadioGroupItem value="2">Option 2</RadioGroupItem>
        </RadioGroup>
        ```
    </Tab>
    <Tab title="Boxed">
        ```tsx
        import {BoxedRadioGroup, RadioGroupItem} from "@/components/ui/Checkbocx";

        <BoxedRadioGroup direction="column">
            <RadioGroupItem value="1">Option 1</RadioGroupItem>
            <RadioGroupItem value="2">Option 2</RadioGroupItem>
        </BoxedRadioGroup>
        ```
    </Tab>
</Tabs>
