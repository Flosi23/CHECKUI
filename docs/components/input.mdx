## Installation

<Note>This Component depends on the [Spinner](/components/spinner) component</Note>

<Steps>
    <Step title={depsOnStepTitle}>
        <Card title="Spinner" href="/components/spinner"></Card>
    </Step>
    <Step title={cssStepTitle}>
        ```css
        input[type="number"]::-webkit-outer-spin-button,
        input[type="number"]::-webkit-inner-spin-button,
        input[type="number"] {
            -webkit-appearance: none;
            margin: 0;
        }
        ```
    </Step>
    <Step title={codeStepTitle}>
        <Input/>
    </Step>
    <Step title={updateStepTitle}></Step>
</Steps>

## Usage

```tsx
import { Input } from "@lib/Input";

<Input label="Name" name="name" />;
```
